cmake_minimum_required (VERSION 2.6)
project (Tutorial)


# The version number.
set (BouncingBalls_VERSION_MAJOR 1)
set (BouncingBalls_VERSION_MINOR 0)

# configure a header file to pass some of the CMake settings
# to the source code
configure_file (
	"${PROJECT_SOURCE_DIR}/BouncingBallsConfig.h.in"
	"${PROJECT_BINARY_DIR}/BouncingBallsConfig.h"
               )

# add the binary tree to the search path for include files
# so that we will find the BouncingBallsConfig.h
include_directories("${PROJECT_BINARY_DIR}")

# add the executable 	
add_executable(BoundcingBalls.cxx)
